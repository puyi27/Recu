4.1. Reproductor con Barra de Progreso Dinámica

<audio controls>
  <source src="audio.mp3" type="audio/mpeg">
</audio>

<div style="width: 500px; background: #222; padding: 10px;">
  <video id="vidReproductor" width="100%">
    <source src="video.mp4" type="video/mp4">
  </video>
  
  <div id="barra-fondo" style="width: 100%; height: 10px; background: gray; cursor: pointer; margin-top: 10px;">
    <div id="barra-progreso" style="height: 100%; width: 0%; background: red;"></div>
  </div>
  
  <button onclick="activarVideo()">Play</button>
  <button onclick="pausarVideo()">Pausar</button>
  <button onclick="reiniciarVideo()">Reiniciar</button>
</div>

<script>
  const vid = document.getElementById("vidReproductor");
  const barraProgreso = document.getElementById("barra-progreso");
  const barraFondo = document.getElementById("barra-fondo");

  function activarVideo() { vid.muted = false; vid.volume = 1.0; vid.play(); }
  function pausarVideo() { vid.pause(); }
  function reiniciarVideo() { vid.currentTime = 0; vid.play(); }

  vid.addEventListener("timeupdate", () => {
    if (!isNaN(vid.duration)) {
        let porcentaje = (vid.currentTime / vid.duration) * 100;
        barraProgreso.style.width = porcentaje + "%";
    }
  });

  barraFondo.addEventListener("click", (e) => {
    let anchoTotal = barraFondo.offsetWidth;
    let clickX = e.offsetX;
    vid.currentTime = (clickX / anchoTotal) * vid.duration;
  });
</script>

  const ctxLikes = document.getElementById("canvasLikes").getContext("2d");
  let totalLikes = 0;
  function dibujarBarra() {
      ctxLikes.clearRect(0, 0, 400, 50);
      ctxLikes.fillStyle = "limegreen";
      ctxLikes.fillRect(0, 0, (totalLikes / 200) * 400, 50);
      ctxLikes.fillStyle = "black";
      ctxLikes.font = "20px Arial";
      ctxLikes.fillText("Likes: " + totalLikes, 10, 32);
  }
  function sumarLike() { if (totalLikes < 200) { totalLikes += 10; dibujarBarra(); } }
</script>

    const gradient = ctxOnda.createLinearGradient(0, 0, 0, 200);

    ctxOnda.moveTo(0, 100);

13. CSS3: Distribución de Pantalla (Grid y Flexbox)

Para escapar del flujo lineal tradicional y estructurar la página por bloques, se recomienda combinar CSS Grid para el layout principal (bidimensional) y Flexbox para alinear elementos internos (unidimensional). Esto proporciona interfaces robustas y completamente responsivas.

13.1. Maquetación Clásica con CSS Grid

El uso de grid-template-areas permite definir zonas de la página asignándoles un nombre, lo que facilita enormemente crear la estructura general (Cabecera, Menú, Contenido y Pie) de un solo vistazo.

Cabecera (Header)
Menú Lateral
Contenido Principal
Pie de página (Footer)
<style>
  .grid-layout {
      display: grid;
      grid-template-areas: 
          "header header"
          "nav main"
          "footer footer";
      grid-template-columns: 250px 1fr;
      grid-template-rows: auto 1fr auto;
      min-height: 100vh;
      gap: 10px;
  }
  .grid-layout header { grid-area: header; background: #333; color: white; }
  .grid-layout nav { grid-area: nav; background: #777; color: white; }
  .grid-layout main { grid-area: main; background: #aaa; }
  .grid-layout footer { grid-area: footer; background: #000; color: white; }
</style>

<div class="grid-layout">
    <header>Cabecera</header>
    <nav>Menú Lateral</nav>
    <main>Contenido Principal</main>
    <footer>Pie de página</footer>
</div>
