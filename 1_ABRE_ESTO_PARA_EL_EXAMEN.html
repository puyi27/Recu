$("#caja_arrastrable").on("dragstart", function(e) {

    e.originalEvent.dataTransfer.setData("mochila", e.target.id);

    $(this).append( document.getElementById(idSacado) );
});

let miLienzo = document.getElementById("pizarra");
let ctx = miLienzo.getContext("2d");
let posicionX = 0; // Donde empieza el cuadrado

function bucleMagico() {
    // 1. BORRAMOS LA PIZARRA ENTERA (OBLIGATORIO)
    ctx.clearRect(0, 0, miLienzo.width, miLienzo.height); 

    // 2. DIBUJAMOS EL CUADRADO AZUL
    ctx.fillStyle = "blue";
    ctx.fillRect(posicionX, 50, 50, 50); // X, Y, Ancho, Alto

    // 3. LE SUMAMOS 2 PARA QUE SE MUEVA A LA DERECHA
    posicionX += 2; 
    
    // Si se sale de la pantalla, que vuelva al principio
    if (posicionX > miLienzo.width) { posicionX = -50; }

    // 4. VUELVE A LLAMARSE A SÍ MISMO (BUCLE INFINITO)
    requestAnimationFrame(bucleMagico); 
}
// Arrancamos el bucle la primera vez
bucleMagico();

🔹 4. Barra Progreso de Vídeo Dinámica

<video id="vidReproductor" width="500"><source src="clip.mp4" type="video/mp4"></video>
<div id="barra-fondo" style="width: 500px; height: 10px; background: gray; cursor: pointer;">
  <div id="barra-progreso" style="height: 100%; width: 0%; background: red;"></div>

<!-- JAVASCRIPT (Pegar abajo del todo, fuera de jQuery) -->
<script>
  const vid = document.getElementById("vidReproductor");
  const barraP = document.getElementById("barra-progreso");
  const barraF = document.getElementById("barra-fondo");

  // La barra avanza sola
  vid.addEventListener("timeupdate", () => {
    let porcentaje = (vid.currentTime / vid.duration) * 100;
    barraP.style.width = porcentaje + "%";
  });

  // Hacer click en la barra adelanta el vídeo
  barraF.addEventListener("click", (e) => {
    let anchoTotal = barraF.offsetWidth;
    let nuevoTiempo = (e.offsetX / anchoTotal) * vid.duration;
    vid.currentTime = nuevoTiempo;
  });
</script>

<script>
  function arrastrar(e) { 
      e.dataTransfer.setData("id_caja", e.target.id); 
  }
  function permitir(e) { 
      e.preventDefault(); // SUPER IMPORTANTE: Quita el bloqueo del navegador
  } 
  function soltar(e) {
      e.preventDefault();
      let id = e.dataTransfer.getData("id_caja");
      e.target.appendChild(document.getElementById(id)); // Mueve la caja dentro
  }
</script>

🔹 18. Canvas Avanzado: Gráfico Circular (Pie Chart)

👇 Dibujado con Radianes (Math.PI) 👇 
<!-- HTML -->
<canvas id="canvasPie" width="200" height="200"></canvas>

<!-- JS FUERA DE JQUERY -->
<script>
  const ctxP = document.getElementById("canvasPie").getContext("2d");
  
  // Fórmula: arc(x, y, radio, anguloInicio, anguloFin)
  // Math.PI es media vuelta (180 grados). Math.PI * 2 es vuelta entera (360)
  
  // Mitad Azul (0 a 180 grados)
  ctxP.beginPath();
  ctxP.moveTo(100, 100); // Mueve el lápiz al centro exacto
  ctxP.arc(100, 100, 100, 0, Math.PI, false); 
  ctxP.fillStyle = "#3498db";
  ctxP.fill();

  // Mitad Roja (180 a 360 grados)
  ctxP.beginPath();
  ctxP.moveTo(100, 100);
  ctxP.arc(100, 100, 100, Math.PI, Math.PI * 2, false); 
  ctxP.fillStyle = "#e74c3c";
  ctxP.fill();
</script>
